50 @BenchmarkMode(Mode.Throughput)
51 @OutputTimeUnit(TimeUnit.MICROSECONDS)
52 @Warmup(iterations = 5, time = 1, timeUnit = TimeUnit.SECONDS)
53 @Measurement(iterations = 5, time = 1, timeUnit = TimeUnit.SECONDS)
54 @Fork(5)
55 public class JMHSample_22_FalseSharing {

57  /*
58  * One of the unusual thing that can bite you back is false sharing.
59  * If two threads access (and possibly modify) the adjacent values
60  * in memory, chances are, they are modifying the values on the same
61  * cache line. This can yield significant (artificial) slowdowns.
62  *
63  * JMH helps you to alleviate this: @States are automatically padded.
64  * This padding does not extend to the State internals though,
65  * as we will see in this example. You have to take care of this on
66  * your own.
67  */
68 
69  /*
70  * Suppose we have two threads:
71  * a) innocuous reader which blindly reads its own field
72  * b) furious writer which updates its own field
73  */
74 
75  /*
76  * BASELINE EXPERIMENT:
77  * Because of the false sharing, both reader and writer will experience
78  * penalties.
79  */
80 
81  @State(Scope.Group)
82  public static class StateBaseline {
83  int readOnly;
84  int writeOnly;
85  }
86 
87  @Benchmark
88  @Group("baseline")
89  public int reader(StateBaseline s) {
90  return s.readOnly;
91  }
92 
93  @Benchmark
94  @Group("baseline")
95  public void writer(StateBaseline s) {
96  s.writeOnly++;
97  }
98 
99  /*
100  * APPROACH 1: PADDING
101  *
102  * We can try to alleviate some of the effects with padding.
103  * This is not versatile because JVMs can freely rearrange the
104  * field order.
105  */
106 
107  @State(Scope.Group)
108  public static class StatePadded {
109  int readOnly;
110  int p01, p02, p03, p04, p05, p06, p07, p08;
111  int p11, p12, p13, p14, p15, p16, p17, p18;
112  int writeOnly;
113  int q01, q02, q03, q04, q05, q06, q07, q08;
114  int q11, q12, q13, q14, q15, q16, q17, q18;
115  }
116 
117  @Benchmark
118  @Group("padded")
119  public int reader(StatePadded s) {
120  return s.readOnly;
121  }
122 
123  @Benchmark
124  @Group("padded")
125  public void writer(StatePadded s) {
126  s.writeOnly++;
127  }
128 
129  /*
130  * APPROACH 2: CLASS HIERARCHY TRICK
131  *
132  * We can alleviate false sharing with this convoluted hierarchy trick,
133  * using the fact that superclass fields are usually laid out first.
134  * In this construction, the protected field will be squashed between
135  * paddings.
136  */
137 
138  public static class StateHierarchy_1 {
139  int readOnly;
140  }
141 
142  public static class StateHierarchy_2 extends StateHierarchy_1 {
143  int p01, p02, p03, p04, p05, p06, p07, p08;
144  int p11, p12, p13, p14, p15, p16, p17, p18;
145  }
146 
147  public static class StateHierarchy_3 extends StateHierarchy_2 {
148  int writeOnly;
149  }
150 
151  public static class StateHierarchy_4 extends StateHierarchy_3 {
152  int q01, q02, q03, q04, q05, q06, q07, q08;
153  int q11, q12, q13, q14, q15, q16, q17, q18;
154  }
155 
156  @State(Scope.Group)
157  public static class StateHierarchy extends StateHierarchy_4 {
158  }
159 
160  @Benchmark
161  @Group("hierarchy")
162  public int reader(StateHierarchy s) {
163  return s.readOnly;
164  }
165 
166  @Benchmark
167  @Group("hierarchy")
168  public void writer(StateHierarchy s) {
169  s.writeOnly++;
170  }
171 
172  /*
173  * APPROACH 3: ARRAY TRICK
174  *
175  * This trick relies on the contiguous allocation of an array.
176  * Instead of placing the fields in the class, we mangle them
177  * into the array at very sparse offsets.
178  */
179 
180  @State(Scope.Group)
181  public static class StateArray {
182  int[] arr = new int[128];
183  }
184 
185  @Benchmark
186  @Group("sparse")
187  public int reader(StateArray s) {
188  return s.arr[0];
189  }
190 
191  @Benchmark
192  @Group("sparse")
193  public void writer(StateArray s) {
194  s.arr[64]++;
195  }
