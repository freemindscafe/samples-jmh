42 import java.util.concurrent.TimeUnit;

44 public class JMHSample_02_BenchmarkModes {
45 
46  /*
47  * JMH generates lots of synthetic code for the benchmarks for you
48  * during the benchmark compilation. JMH can measure the benchmark
49  * methods in lots of modes. Users may select the default benchmark
50  * mode with the special annotation, or select/override the mode via
51  * the runtime options.
52  *
53  * With this scenario, we start to measure something useful. Note that
54  * our payload code potentially throws the exceptions, and we can just
55  * declare them to be thrown. If the code throws the actual exception,
56  * the benchmark execution will stop with error.
57  *
58  * When you are puzzled with some particular behavior, it usually helps
59  * to look into the generated code. You might see the code is doing not
60  * something you intend it to do. Good experiments always follow up on
61  * the experimental setup, and cross-checking the generated code is an
62  * important part of that follow up.
63  *
64  * The generated code for this particular sample is somewhere at
65  * target/generated-sources/annotations/.../JMHSample_02_BenchmarkModes.java
66  */
67 
68  /*
69  * Mode.Throughput, as stated in its Javadoc, measures the raw throughput
70  * by continuously calling the benchmark method in a time-bound iteration,
71  * and counting how many times we executed the method.
72  *
73  * We are using the special annotation to select the units to measure in,
74  * although you can use the default.
75  */
76 
77  @Benchmark
78  @BenchmarkMode(Mode.Throughput)
79  @OutputTimeUnit(TimeUnit.SECONDS)
80  public void measureThroughput() throws InterruptedException {
81  TimeUnit.MILLISECONDS.sleep(100);
82  }
83 
84  /*
85  * Mode.AverageTime measures the average execution time, and it does it
86  * in the way similar to Mode.Throughput.
87  *
88  * Some might say it is the reciprocal throughput, and it really is.
89  * There are workloads where measuring times is more convenient though.
90  */
91 
92  @Benchmark
93  @BenchmarkMode(Mode.AverageTime)
94  @OutputTimeUnit(TimeUnit.MICROSECONDS)
95  public void measureAvgTime() throws InterruptedException {
96  TimeUnit.MILLISECONDS.sleep(100);
97  }
98 
99  /*
100  * Mode.SampleTime samples the execution time. With this mode, we are
101  * still running the method in a time-bound iteration, but instead of
102  * measuring the total time, we measure the time spent in *some* of
103  * the * benchmark method calls.
104  *
105  * This allows us to infer the distributions, percentiles, etc.
106  *
107  * JMH also tries to auto-adjust sampling frequency: if the method
108  * is long enough, you will end up capturing all the samples.
109  */
110  @Benchmark
111  @BenchmarkMode(Mode.SampleTime)
112  @OutputTimeUnit(TimeUnit.MICROSECONDS)
113  public void measureSamples() throws InterruptedException {
114  TimeUnit.MILLISECONDS.sleep(100);
115  }
116 
117  /*
118  * Mode.SingleShotTime measures the single method invocation time. As the Javadoc
119  * suggests, we do only the single benchmark method invocation. The iteration
120  * time is meaningless in this mode: as soon as benchmark method stops, the
121  * iteration is over.
122  *
123  * This mode is useful to do cold startup tests, when you specifically
124  * do not want to call the benchmark method continuously.
125  */
126 
127  @Benchmark
128  @BenchmarkMode(Mode.SingleShotTime)
129  @OutputTimeUnit(TimeUnit.MICROSECONDS)
130  public void measureSingleShot() throws InterruptedException {
131  TimeUnit.MILLISECONDS.sleep(100);
132  }
133 
134  /*
135  * We can also ask for multiple benchmark modes at once. All the tests
136  * above can be replaced with just a single test like this:
137  */
138 
139  @Benchmark
140  @BenchmarkMode({Mode.Throughput, Mode.AverageTime, Mode.SampleTime, Mode.SingleShotTime})
141  @OutputTimeUnit(TimeUnit.MICROSECONDS)
142  public void measureMultiple() throws InterruptedException {
143  TimeUnit.MILLISECONDS.sleep(100);
144  }
145 
146  /*
147  * Or even...
148  */
149 
150  @Benchmark
151  @BenchmarkMode(Mode.All)
152  @OutputTimeUnit(TimeUnit.MICROSECONDS)
153  public void measureAll() throws InterruptedException {
154  TimeUnit.MILLISECONDS.sleep(100);
155  }
156 
157  /*
158  * ============================== HOW TO RUN THIS TEST: ====================================
159  *
160  * You are expected to see the different run modes for the same benchmark.
161  * Note the units are different, scores are consistent with each other.
162  *
163  * You can run this test:
164  *
165  * a) Via the command line:
166  * $ mvn clean install
167  * $ java -jar target/benchmarks.jar JMHSample_02 -wi 5 -i 5 -f 1
168  * (we requested 5 warmup/measurement iterations, single fork)
169  *
170  * b) Via the Java API:
171  * (see the JMH homepage for possible caveats when running from IDE:
172  * http://openjdk.java.net/projects/code-tools/jmh/)
173  */
174 
175  public static void main(String[] args) throws RunnerException {
176  Options opt = new OptionsBuilder()
177  .include(JMHSample_02_BenchmarkModes.class.getSimpleName())
178  .warmupIterations(5)
179  .measurementIterations(5)
180  .forks(1)
181  .build();
182 
183  new Runner(opt).run();
184  }
